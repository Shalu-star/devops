##### Terraform setup #######

wget https://releases.hashicorp.com/terraform/1.4.6/terraform_1.4.6_linux_amd64.zip
sudo apt-get install unzip
unzip terraform_1.4.6_linux_amd64.zip
sudo cp terraform /usr/bin/
terraform -v
apt-get remove terraform -y
echo $PATH
terraform -help

terraform init
terraform apply
terraform destroy
